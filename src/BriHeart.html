top_ten <- heart_data %>%

  mutate(quarter_one = quantile(heart_rate, .25), quarter_three = quantile(heart_rate, .75), iqr = quarter_three - quarter_one) %>%
  filter(heart_rate > quarter_one - (1.5 * iqr),
